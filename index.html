Grafos en Java

Código completo con clases separadas: Nodo, Arista, Grafo y Main.

Nodo<V> Arista<V> Grafo<V> Main
↓ Desplázate para explorar
🏗️

Arquitectura

El proyecto se divide en 4 archivos .java independientes:

📌 Nodo.java

Vértice genérico. Guarda el dato y su lista de aristas.

🔗 Arista.java

Conexión con origen, destino y peso (1.0 por defecto).

🗂️ Grafo.java

Map de nodos, CRUD, BFS, DFS.

▶️ Main.java

Punto de entrada: crea el grafo y lo usa.

📌 Nodo.java
import java.util.*;
import java.util.Objects;

/**
 * Clase NODO (vértice) del grafo.
 * Genérica: puede guardar cualquier tipo de dato.
 */
public class Nodo<V> {

    private V dato;
    private List<Arista<V>> aristas;

    // ── CONSTRUCTOR ──
    public Nodo(V dato) {
        this.dato = dato;
        this.aristas = new ArrayList<>();
    }

    // ── GETTERS ──
    public V getDato() {
        return this.dato;
    }

    public List<Arista<V>> getAristas() {
        return this.aristas;
    }

    // ── AGREGAR ARISTA DESDE ESTE NODO ──
    public void agregarArista(Nodo<V> destino, double peso) {
        this.aristas.add(new Arista<>(this.dato, destino.getDato(), peso));
    }

    // ── EQUALS Y HASHCODE (basados en el dato) ──
    @Override
    public boolean equals(Object obj) {
        if (this.dato == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
        return Objects.equals(this.dato, ((Nodo<?>) obj).dato);
    }

    @Override
    public int hashCode() {
        return Objects.hashCode(this.dato);
    }

    @Override
    public String toString() {
        return "Nodo(" + this.dato + ")";
    }
}
🔗 Arista.java
import java.util.Objects;

/**
 * Clase ARISTA (edge) del grafo.
 * Conecta origen -> destino con un peso numérico.
 * Para grafos no ponderados, peso = 1.0 por defecto.
 */
public class Arista<V> {

    private V origen;
    private V destino;
    private double peso;

    // ── CONSTRUCTOR COMPLETO ──
    public Arista(V origen, V destino, double peso) {
        this.origen = origen;
        this.destino = destino;
        this.peso = peso;
    }

    // ── CONSTRUCTOR SIN PESO (asume 1.0) ──
    public Arista(V origen, V destino) {
        this(origen, destino, 1.0);
    }

    // ── GETTERS ──
    public V getOrigen() {
        return this.origen;
    }

    public V getDestino() {
        return this.destino;
    }

    public double getPeso() {
        return this.peso;
    }

    // ── SETTER ──
    public void setPeso(double peso) {
        this.peso = peso;
    }

    // ── EQUALS ──
    @Override
    public boolean equals(Object obj) {
        if (obj == null || getClass() != obj.getClass()) return false;
        Arista<?> otro = (Arista<?>) obj;
        return Objects.equals(this.origen, otro.origen)
            && Objects.equals(this.destino, otro.destino);
    }

    @Override
    public String toString() {
        return this.origen + " -> " + this.destino
            + " (peso: " + this.peso + ")";
    }
}
🗂️ Grafo.java
import java.util.*;

/**
 * Implementación de GRAFO con lista de adyacencia.
 * Mapa: dato del vértice -> objeto Nodo.
 */
public class Grafo<V> {

    private Map<V, Nodo<V>> nodos;
    private boolean dirigido;

    // ── CONSTRUCTOR ──
    public Grafo(boolean dirigido) {
        this.dirigido = dirigido;
        this.nodos = new HashMap<>();
    }

    // ── CREAR NODO ──
    public void agregarNodo(V dato) {
        this.nodos.putIfAbsent(dato, new Nodo<>(dato));
    }

    // ── CREAR ARISTA CON PESO ──
    public void agregarArista(V origen, V destino, double peso) {
        this.agregarNodo(origen);
        this.agregarNodo(destino);
        // Arista en un sentido
        this.nodos.get(origen).agregarArista(this.nodos.get(destino), peso);
        // Si NO es dirigido: agregar arista inversa
        if (!this.dirigido) {
            this.nodos.get(destino).agregarArista(this.nodos.get(origen), peso);
        }
    }

    // ── CREAR ARISTA SIN PESO (peso = 1.0) ──
    public void agregarArista(V origen, V destino) {
        this.agregarArista(origen, destino, 1.0);
    }

    // ── ELIMINAR NODO ──
    public void eliminarNodo(V dato) {
        if (this.nodos.containsKey(dato)) {
            // Quitar todas las aristas que apuntan a este nodo
            for (Nodo<V> n : this.nodos.values()) {
                n.getAristas().removeIf(a -> a.getDestino().equals(dato));
            }
            this.nodos.remove(dato);
        }
    }

    // ── CONSULTAS ──
    public boolean existeNodo(V dato) {
        return this.nodos.containsKey(dato);
    }

    public boolean existeArista(V origen, V destino) {
        Nodo<V> n = this.nodos.get(origen);
        if (n == null) return false;
        return n.getAristas().stream()
            .anyMatch(a -> a.getDestino().equals(destino));
    }

    public int grado(V dato) {
        Nodo<V> n = this.nodos.get(dato);
        return n != null ? n.getAristas().size() : 0;
    }

    public List<V> obtenerVecinos(V dato) {
        Nodo<V> n = this.nodos.get(dato);
        if (n == null) return Collections.emptyList();
        return n.getAristas().stream()
            .map(Arista::getDestino).toList();
    }

    public int getNumNodos() {
        return this.nodos.size();
    }

    // ── BFS (recorrido en anchura) ──
    public List<V> bfs(V inicio) {
        List<V> resultado = new ArrayList<>();
        Set<V> visitados = new HashSet<>();
        Queue<V> cola = new LinkedList<>();
        cola.offer(inicio);
        visitados.add(inicio);

        while (!cola.isEmpty()) {
            V actual = cola.poll();
            resultado.add(actual);
            for (V vecino : this.obtenerVecinos(actual)) {
                if (!visitados.contains(vecino)) {
                    visitados.add(vecino);
                    cola.offer(vecino);
                }
            }
        }
        return resultado;
    }

    // ── DFS (recorrido en profundidad) ──
    public List<V> dfs(V inicio) {
        List<V> resultado = new ArrayList<>();
        this.dfsRec(inicio, new HashSet<>(), resultado);
        return resultado;
    }

    private void dfsRec(V actual, Set<V> visitados, List<V> resultado) {
        visitados.add(actual);
        resultado.add(actual);
        for (V vecino : this.obtenerVecinos(actual)) {
            if (!visitados.contains(vecino)) {
                this.dfsRec(vecino, visitados, resultado);
            }
        }
    }

    // ── IMPRIMIR EL GRAFO ──
    public void imprimir() {
        System.out.println("=== "
            + (this.dirigido ? "Grafo Dirigido" : "Grafo No Dirigido")
            + " ===");
        for (Nodo<V> n : this.nodos.values()) {
            System.out.print("  " + n.getDato() + " -> ");
            if (n.getAristas().isEmpty()) {
                System.out.println("[]");
            } else {
                System.out.println(n.getAristas());
            }
        }
    }
}
▶️ Main.java
/**
 * Punto de entrada del programa.
 * Crea el grafo, lo pobla con datos y ejecuta operaciones.
 */
public class Main {

    public static void main(String[] args) {

        // ══ PASO 1: Crear grafo NO DIRIGIDO ══
        Grafo<String> grafo = new Grafo<>(false);

        // ══ PASO 2: Agregar aristas ══
        // (los nodos se crean automáticamente)
        grafo.agregarArista("A", "B");
        grafo.agregarArista("A", "D");
        grafo.agregarArista("B", "C");
        grafo.agregarArista("B", "E");
        grafo.agregarArista("D", "E");
        grafo.agregarArista("C", "F");
        grafo.agregarArista("E", "F");

        // ══ PASO 3: Imprimir el grafo ══
        grafo.imprimir();

        // ══ PASO 4: Consultas ══
        System.out.println("\nNodos totales: "
            + grafo.getNumNodos());
        System.out.println("Existe nodo B? "
            + grafo.existeNodo("B"));
        System.out.println("Existe arista A->B? "
            + grafo.existeArista("A", "B"));
        System.out.println("Grado de B: "
            + grafo.grado("B"));
        System.out.println("Vecinos de B: "
            + grafo.obtenerVecinos("B"));

        // ══ PASO 5: Recorridos ══
        System.out.println("\nBFS desde A: "
            + grafo.bfs("A"));
        System.out.println("DFS desde A: "
            + grafo.dfs("A"));
    }
}

📖 ¿Cómo funciona la separación?

  1. Nodo.java — Guarda el dato y su lista de aristas. Genérico para cualquier tipo.
  2. Arista.java — Guarda origen, destino y peso. Sin peso usa 1.0.
  3. Grafo.java — Tiene Map<V, Nodo<V>> y el flag dirigido. Si es false, duplica cada arista.
  4. Main.java — Crea new Grafo<>(false), agrega aristas y llama a los métodos.

Grafo Dirigido

Se reutilizan las mismas 3 clases. Solo cambias a new Grafo<>(true).

Flechas = dirección. A->B no implica B->A

▶️ MainDirigido.java
/**
 * Main para grafo DIRIGIDO.
 * Usa las mismas clases: Nodo, Arista, Grafo.
 */
public class MainDirigido {

    public static void main(String[] args) {

        // TRUE = dirigido (aristas en un solo sentido)
        Grafo<String> g = new Grafo<>(true);

        // Red de páginas web: links entre sitios
        g.agregarArista("Google", "YouTube");
        g.agregarArista("Google", "Gmail");
        g.agregarArista("YouTube", "Google");
        g.agregarArista("Gmail", "Drive");
        g.agregarArista("YouTube", "Drive");

        g.imprimir();

        System.out.println("\nGoogle -> YouTube: "
            + g.existeArista("Google", "YouTube"));
        System.out.println("YouTube -> Google: "
            + g.existeArista("YouTube", "Google"));
        System.out.println("Drive -> Gmail: "
            + g.existeArista("Drive", "Gmail"));
    }
}

🔑 La única diferencia

new Grafo<>(true) vs new Grafo<>(false). La clase Grafo internamente decide si duplicar la arista o no. No tocas Nodo ni Arista.

Grafo Cíclico

Contiene al menos un ciclo: un camino que vuelve al nodo de inicio. Se detecta con DFS.

Ciclo A->B->C->A (líneas gruesas rojas)

▶️ MainCiclico.java
import java.util.*;

/**
 * Main para grafo CÍCLICO.
 * Incluye detección de ciclos con DFS.
 */
public class MainCiclico {

    public static void main(String[] args) {

        // Dirigido para ver mejor el ciclo
        Grafo<String> g = new Grafo<>(true);

        // Crear ciclo: A -> B -> C -> A
        g.agregarArista("A", "B");
        g.agregarArista("B", "C");
        g.agregarArista("C", "A");  // Cierra el ciclo!
        g.agregarArista("B", "E");
        g.agregarArista("A", "D");
        g.agregarArista("D", "E");
        g.agregarArista("E", "F");
        g.agregarArista("C", "F");
        g.agregarArista("F", "D");  // Segundo ciclo: D-E-F-D

        g.imprimir();

        System.out.println("\nTiene ciclo? "
            + tieneCiclo(g));
    }

    // ── Detectar ciclo con DFS ──
    private static boolean tieneCiclo(Grafo<String> g) {
        Set<String> vis = new HashSet<>();
        Set<String> stk = new HashSet<>();

        // Recorrer todos los nodos del grafo
        for (String v : g.obtenerVecinos("A")) {
            if (dfsCiclo(g, v, vis, stk)) {
                return true;
            }
        }
        return false;
    }

    private static boolean dfsCiclo(Grafo<String> g,
            String act, Set<String> vis, Set<String> stk) {
        vis.add(act);
        stk.add(act);  // Marcar en la pila actual

        for (String vec : g.obtenerVecinos(act)) {
            if (!vis.contains(vec)) {
                // Si no visitado, seguir explorando
                if (dfsCiclo(g, vec, vis, stk)) {
                    return true;
                }
            } else if (stk.contains(vec)) {
                // Vecino ya está en la pila = ciclo!
                return true;
            }
        }
        stk.remove(act);  // Sacar de la pila al terminar
        return false;
    }
}

🔑 ¿Cómo detecta DFS un ciclo?

Usa dos conjuntos: visitados (nodos ya vistos) y stk (nodos en la ruta actual de DFS). Si encuentra un vecino que ya está en stk, significa que volvió a un nodo del camino actual = ciclo detectado.

🌳

Grafo Acíclico (DAG)

Sin ciclos. Permite ordenamiento topológico: ordenar nodos respetando dependencias.

DAG: todas las flechas van "hacia abajo". Imposible volver al inicio.

▶️ MainAciclico.java
import java.util.*;

/**
 * Main para grafo ACÍCLICO (DAG).
 * Incluye ordenamiento topológico con DFS.
 */
public class MainAciclico {

    public static void main(String[] args) {

        Grafo<String> g = new Grafo<>(true);

        // Dependencias de un proyecto
        g.agregarArista("Planificar", "Disenar");
        g.agregarArista("Planificar", "Investigar");
        g.agregarArista("Disenar", "Programar");
        g.agregarArista("Investigar", "Programar");
        g.agregarArista("Programar", "Probar");
        g.agregarArista("Programar", "Documentar");
        g.agregarArista("Probar", "Lanzar");
        g.agregarArista("Documentar", "Lanzar");

        g.imprimir();

        System.out.println("\nEs aciclico? "
            + !tieneCiclo(g));

        System.out.println("\nOrden topologico:");
        for (String t : ordenTopologico(g)) {
            System.out.println("  " + t);
        }
    }

    // ── Orden topológico con DFS ──
    private static List<String> ordenTopologico(Grafo<String> g) {
        Set<String> vis = new HashSet<>();
        Deque<String> pila = new ArrayDeque<>();

        // Recorrer todos los vecinos del nodo inicial
        for (String v : g.obtenerVecinos("Planificar")) {
            if (!vis.contains(v)) {
                dfsTop(g, v, vis, pila);
            }
        }
        pila.push("Planificar");

        // La pila contiene el orden topológico
        List<String> res = new ArrayList<>();
        while (!pila.isEmpty()) {
            res.add(pila.pop());
        }
        return res;
    }

    // DFS que apila al terminar (post-order)
    private static void dfsTop(Grafo<String> g,
            String act, Set<String> vis, Deque<String> pila) {
        vis.add(act);
        for (String vec : g.obtenerVecinos(act)) {
            if (!vis.contains(vec)) {
                dfsTop(g, vec, vis, pila);
            }
        }
        pila.push(act);  // Se apila al terminar
    }

    // ── Reutilizamos detección de ciclo ──
    private static boolean tieneCiclo(Grafo<String> g) {
        Set<String> vis = new HashSet<>();
        Set<String> stk = new HashSet<>();
        for (String v : g.obtenerVecinos("Planificar")) {
            if (dfsCiclo(g, v, vis, stk)) return true;
        }
        return false;
    }

    private static boolean dfsCiclo(Grafo<String> g,
            String act, Set<String> vis, Set<String> stk) {
        vis.add(act);
        stk.add(act);
        for (String vec : g.obtenerVecinos(act)) {
            if (!vis.contains(vec)) {
                if (dfsCiclo(g, vec, vis, stk)) return true;
            } else if (stk.contains(vec)) return true;
        }
        stk.remove(act);
        return false;
    }
}
