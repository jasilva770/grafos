Grafos en Java

Guía completa con código bien estructurado: Nodo, Arista y Main en clases separadas.

Nodo<V> Arista<V> Grafo<V> Main
Desplázate para explorar ↓
🏗️

Arquitectura del Código

Todo el proyecto de grafos se divide en 4 archivos Java separados, siguiendo buenas prácticas de POO y el principio de responsabilidad única:

Cada clase tiene su propio archivo .java. Main crea instancias de las otras tres.

📌 Nodo.java

Representa un vértice. Contiene el dato genérico y la lista de aristas que salen de él.

🔗 Arista.java

Conexión entre dos nodos. Guarda el destino y el peso (por defecto 1 para no ponderados).

🗂️ Grafo.java

La estructura principal. Mapa de nodos, métodos para agregar/eliminar, y algoritmos (BFS, DFS, Dijkstra, etc).

▶️ Main.java

Punto de entrada. Crea el grafo, lo pobla con datos y ejecuta los algoritmos.

🏗️

Grafo Básico — Código Completo Separado

Grafo no dirigido y no ponderado. Cada arista tiene peso implícito de 1. Las conexiones son bidireccionales.

Grafo básico no dirigido: 6 vértices, 7 aristas
📌 Archivo 1 — Nodo.java 
import java.util.*;
import java.util.Objects;

/**
 * Representa un VÉRTICE (nodo) del grafo.
 * Es genérico: puede almacenar cualquier tipo de dato (String, Integer, etc).
 */
public class Nodo<V> {

    private V dato;
    private List<Arista<V>> aristas;

    public Nodo(V dato) {
        dato = dato;
        aristas = new ArrayList<>();
    }

    // ── GETTERS / SETTERS ──
    public V getDato() {
        return dato;
    }

    public List<Arista<V>> getAristas() {
        return aristas;
    }

    // ── MÉTODOS DE UTILIDAD ──
    public void agregarArista(Nodo<V> destino, double peso) {
        aristas.add(new Arista<>(dato, destino.getDato(), peso));
    }

    // Dos nodos son iguales si su dato es igual
    @Override
    public boolean equals(Object obj) {
        if (dato == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
        Nodo<?> other = (Nodo<?>) obj;
        return Objects.equals(dato, other.dato);
    }

    @Override
    public int hashCode() {
        return Objects.hashCode(dato);
    }

    @Override
    public String toString() {
        return String.format("Nodo(%s, aristas=%d)", dato, aristas.size());
    }
}

import java.util.Objects;

/**
 * Representa una ARISTA (edge) del grafo.
 * Conecta un nodo origen con un nodo destino y tiene un peso.
 * Para grafos no ponderados, el peso siempre es 1.0.
 */
public class Arista<V> {

    private V origen;
    private V destino;
    private double peso;

    public Arista(V origen, V destino, double peso) {
        origen = origen;
        destino = destino;
        peso = peso;
    }

    // Constructor sin peso: asume 1.0 (grafo no ponderado)
    public Arista(V origen, V destino) {
        origen, destino, 1.0);
    }

    // ── GETTERS ──
    public V getOrigen() {
        return origen;
    }

    public V getDestino() {
        return destino;
    }

    public double getPeso() {
        return peso;
    }

    // ── SETTER ──
    public void setPeso(double peso) {
        peso = peso;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj == null) return false;
        if (getClass() != obj.getClass()) return false;
        Arista<?> other = (Arista<?>) obj;
        return Objects.equals(origen, other.origen)
            && Objects.equals(destino, other.destino);
    }

    @Override
    public String toString() {
        return String.format("%s → %s (peso: %.1f)", origen, destino, peso);
    }
}
🗂️ Archivo 3 — Grafo.java 
import java.util.*;

/**
 * Implementación de un GRAFO usando lista de adyacencia.
 * Cada nodo se almacena en un HashMap para acceso O(1).
 */
public class Grafo<V> {

    private Map<V, Nodo<V>> nodos;
    private boolean dirigido;

    public Grafo(boolean dirigido) {
        dirigido = dirigido;
        nodos = new HashMap<>();
    }

    // ── CRUD BÁSICO ──
    public void agregarNodo(V dato) {
        if (!nodos.containsKey(dato)) {
            nodos.put(dato, new Nodo<>(dato));
        }
    }

    public void agregarArista(V origen, V destino, double peso) {
        agregarNodo(origen);
        agregarNodo(destino);
        nodos.get(origen).agregarArista(nodos.get(destino), peso);
        if (!dirigido) {
            nodos.get(destino).agregarArista(nodos.get(origen), peso);
        }
    }

    public void agregarArista(V origen, V destino) {
        agregarArista(origen, destino, 1.0);
    }

    public void eliminarNodo(V dato) {
        if (nodos.containsKey(dato)) {
            // Eliminar todas las aristas que apuntan a este nodo
            for (Nodo<V> n : nodos.values()) {
                n.getAristas().removeIf(a -> a.getDestino().equals(dato));
            }
            nodos.remove(dato);
        }
    }

    // ── CONSULTAS ──
    public boolean existeNodo(V dato) {
        return nodos.containsKey(dato);
    }

    public boolean existeArista(V origen, V destino) {
        Nodo<V> n = nodos.get(origen);
        if (n == null) return false;
        return n.getAristas().stream()
            .anyMatch(a -> a.getDestino().equals(destino));
    }

    public int grado(V dato) {
        Nodo<V> n = nodos.get(dato);
        return n != null ? n.getAristas().size() : 0;
    }

    public List<V> obtenerVecinos(V dato) {
        Nodo<V> n = nodos.get(dato);
        if (n == null) return Collections.emptyList();
        return n.getAristas().stream()
            .map(Arista::getDestino)
            .toList();
    }

    public int getNumNodos() { return nodos.size(); }
    public boolean esDirigido() { return dirigido; }

    // ── RECORRIDOS ──
    public List<V> bfs(V inicio) {
        List<V> resultado = new ArrayList<>();
        Set<V> visitados = new HashSet<>();
        Queue<V> cola = new LinkedList<>();

        cola.offer(inicio);
        visitados.add(inicio);

        while (!cola.isEmpty()) {
            V actual = cola.poll();
            resultado.add(actual);
            for (V vecino : obtenerVecinos(actual)) {
                if (!visitados.contains(vecino)) {
                    visitados.add(vecino);
                    cola.offer(vecino);
                }
            }
        }
        return resultado;
    }

    public List<V> dfs(V inicio) {
        List<V> resultado = new ArrayList<>();
        Set<V> visitados = new HashSet<>();
        dfsRecursivo(inicio, visitados, resultado);
        return resultado;
    }

    private void dfsRecursivo(V actual, Set<V> visitados, List<V> resultado) {
        visitados.add(actual);
        resultado.add(actual);
        for (V vecino : obtenerVecinos(actual)) {
            if (!visitados.contains(vecino)) {
                dfsRecursivo(vecino, visitados, resultado);
            }
        }
    }

    // ── IMPRESIÓN ──
    public void imprimir() {
        System.out.println("\\n=== " + (dirigido ? "Grafo Dirigido" : "Grafo No Dirigido") + " ===");
        for (Nodo<V> nodo : nodos.values()) {
            System.out.print("  " + nodo.getDato() + " → ");
            if (nodo.getAristas().isEmpty()) {
                System.out.println("[]");
            } else {
                System.out.println(nodo.getAristas());
            }
        }
    }
}

import java.util.*;

/**
 * Punto de entrada del programa.
 * Crea el grafo, lo pobla y ejecuta los algoritmos.
 */
public class Main {

    public static void main(String[] args) {
        
        // ── 1. CREAR EL GRAFO (no dirigido, no ponderado) ──
        Grafo<String> grafo = new Grafo<>(false);

        // ── 2. AGREGAR ARISTAS (los nodos se crean automáticamente) ──
        grafo.agregarArista("A", "B");
        grafo.agregarArista("A", "D");
        grafo.agregarArista("B", "C");
        grafo.agregarArista("B", "E");
        grafo.agregarArista("D", "E");
        grafo.agregarArista("C", "F");
        grafo.agregarArista("E", "F");

        // ── 3. IMPRIMIR ──
        grafo.imprimir();

        // ── 4. CONSULTAS ──
        System.out.println("\\nTotal de nodos: " + grafo.getNumNodos());
        System.out.println("¿Existe nodo B? " + grafo.existeNodo("B"));
        System.out.println("¿Existe arista A→B? " + grafo.existeArista("A", "B"));
        System.out.println("Grado de B: " + grafo.grado("B"));
        System.out.println("Vecinos de B: " + grafo.obtenerVecinos("B"));

        // ── 5. RECORRIDOS ──
        System.out.println("\\nBFS desde A: " + grafo.bfs("A"));
        System.out.println("DFS desde A: " + grafo.dfs("A"));
    }
}

Grafo Dirigido

Se usa la misma arquitectura (Nodo, Arista, Grafo). La única diferencia es que el Grafo se crea con true para indicar que es dirigido. Las aristas van solo en un sentido.

Grafo dirigido: las flechas indican que la conexión es solo en una dirección
▶️ MainDirigido.java 
public class MainDirigido {

    public static void main(String[] args) {
        
        // ═══════════════════════════════════════════════
        // Se reutilizan las 3 clases: Nodo, Arista, Grafo
        // Solo cambiamos el parámetro a TRUE = dirigido
        // ═══════════════════════════════════════════════
        Grafo<String> grafo = new Grafo<>(true);  // ← TRUE = dirigido

        // Red de páginas web: un link va de A a B, no necesariamente al revés
        grafo.agregarArista("Google", "YouTube");
        grafo.agregarArista("Google", "Gmail");
        grafo.agregarArista("YouTube", "Google");
        grafo.agregarArista("Gmail", "Drive");
        grafo.agregarArista("YouTube", "Drive");

        grafo.imprimir();

        System.out.println("\\n--- Demo: la dirección importa ---");
        System.out.println("Google → YouTube: " + grafo.existeArista("Google", "YouTube"));  // true
        System.out.println("YouTube → Google: " + grafo.existeArista("YouTube", "Google"));  // true
        System.out.println("Drive → Gmail: "    + grafo.existeArista("Drive", "Gmail"));     // false!
    }
}

Grafo Ponderado — Con Pesos

Las 3 clases (Nodo, Arista, Grafo) no cambian. La clase Arista ya soporta pesos. Solo hay que pasar el peso al método agregarArista y usar Dijkstra para caminos más cortos.

Grafo ponderado: cada arista tiene un peso (número amarillo)
▶️ MainPonderado.java 
import java.util.*;

public class MainPonderado {

    public static void main(String[] args) {
        
        Grafo<String> grafo = new Grafo<>(false);

        // Distancias entre ciudades (en km)
        grafo.agregarArista("Madrid", "Barcelona", 621);
        grafo.agregarArista("Madrid", "Valencia", 355);
        grafo.agregarArista("Madrid", "Sevilla", 534);
        grafo.agregarArista("Barcelona", "Valencia", 349);
        grafo.agregarArista("Sevilla", "Valencia", 650);
        grafo.agregarArista("Barcelona", "Sevilla", 990);

        grafo.imprimir();

        // Dijkstra: camino más corto con pesos
        System.out.println("\\nCamino más corto desde Madrid:");
        Map<String, Double> distancias = dijkstra(grafo, "Madrid");
        distancias.forEach((ciudad, dist) ->
            System.out.println("  " + ciudad + ": " + dist + " km"));
    }

    // ── Algoritmo de Dijkstra (externo al Grafo para este ejemplo) ──
    private static Map<String, Double> dijkstra(Grafo<String> g, String inicio) {
        Map<String, Double> dists = new HashMap<>();
        Set<String> visitados = new HashSet<>();

        for (String v : g.obtenerVecinos(inicio)) {
            dists.put(v, Double.POSITIVE_INFINITY);
        }
        dists.put(inicio, 0.0);

        PriorityQueue<Map.Entry<String, Double>> cola =
            new PriorityQueue<>(Comparator.comparingDouble(Map.Entry::getValue));
        cola.add(new AbstractMap.SimpleEntry<>(inicio, 0.0));

        while (!cola.isEmpty()) {
            String actual = cola.poll().getKey();
            if (visitados.contains(actual)) continue;
            visitados.add(actual);

            for (String vecino : g.obtenerVecinos(actual)) {
                double peso = obtenerPeso(g, actual, vecino);
                double nuevaDist = dists.get(actual) + peso;
                if (nuevaDist < dists.getOrDefault(vecino, Double.POSITIVE_INFINITY)) {
                    dists.put(vecino, nuevaDist);
                    cola.add(new AbstractMap.SimpleEntry<>(vecino, nuevaDist));
                }
            }
        }
        return dists;
    }

    private static double obtenerPeso(Grafo<String> g, String a, String b) {
        // Recorre las aristas del nodo 'a' para encontrar la de destino 'b'
        return 1.0; // Simplificado para el ejemplo
    }
}

Grafo Cíclico

Un grafo con al menos un ciclo: un camino que empieza y termina en el mismo nodo. Se usa el mismo código base y se agrega un método para detectar ciclos.

El ciclo A→B→C→A (líneas rojas gruesas) permite volver al inicio
▶️ MainCiclico.java 
import java.util.*;

public class MainCiclico {

    public static void main(String[] args) {
        
        Grafo<String> grafo = new Grafo<>(true);  // Dirigido para ver mejor el ciclo

        // Crear ciclo: A → B → C → A
        grafo.agregarArista("A", "B");
        grafo.agregarArista("B", "C");
        grafo.agregarArista("C", "A");   // ← ¡Esto cierra el ciclo!
        grafo.agregarArista("B", "E");
        grafo.agregarArista("A", "D");
        grafo.agregarArista("D", "E");
        grafo.agregarArista("E", "F");
        grafo.agregarArista("C", "F");
        grafo.agregarArista("F", "D");   // ← Otro ciclo: D→E→F→D

        grafo.imprimir();

        System.out.println("\\n¿Tiene ciclo? " + tieneCiclo(grafo));
    }

    // ── Detección de ciclo con DFS ──
    private static boolean tieneCiclo(Grafo<String> g) {
        Set<String> visitados = new HashSet<>();
        Set<String> enStack = new HashSet<>();

        for (String v : g.obtenerVecinos("A")) {
            if (dfsCiclo(g, v, visitados, enStack)) return true;
        }
        return false;
    }

    private static boolean dfsCiclo(Grafo<String> g, String actual,
                                   Set<String> visitados, Set<String> enStack) {
        visitados.add(actual);
        enStack.add(actual);

        for (String vecino : g.obtenerVecinos(actual)) {
            if (!visitados.contains(vecino)) {
                if (dfsCiclo(g, vecino, visitados, enStack)) return true;
            } else if (enStack.contains(vecino)) {
                return true;  // ¡Ciclo detectado!
            }
        }
        enStack.remove(actual);
        return false;
    }
}

Grafo Acíclico (DAG)

Grafo sin ciclos. No existe ningún camino que regrese al nodo de inicio. Se le llama DAG (Directed Acyclic Graph). Permite hacer ordenamiento topológico.

DAG: todas las flechas van "hacia abajo". Imposible volver al inicio.
▶️ MainAciclico.java 
import java.util.*;

public class MainAciclico {

    public static void main(String[] args) {
        
        Grafo<String> grafo = new Grafo<>(true);

        // Dependencias de un proyecto de software
        grafo.agregarArista("Planificar", "Diseñar");
        grafo.agregarArista("Planificar", "Investigar");
        grafo.agregarArista("Diseñar", "Programar");
        grafo.agregarArista("Investigar", "Programar");
        grafo.agregarArista("Programar", "Probar");
        grafo.agregarArista("Programar", "Documentar");
        grafo.agregarArista("Probar", "Lanzar");
        grafo.agregarArista("Documentar", "Lanzar");

        grafo.imprimir();

        System.out.println("\\n¿Es acíclico? " + !tieneCiclo(grafo));

        System.out.println("\\nOrden topológico (orden de ejecución):");
        List<String> orden = ordenTopologico(grafo);
        for (int i = 0; i < orden.size(); i++) {
            System.out.println("  Paso " + (i+1) + ": " + orden.get(i));
        }
    }

    // ── Ordenamiento Topológico con DFS ──
    private static List<String> ordenTopologico(Grafo<String> g) {
        Set<String> visitados = new HashSet<>();
        Deque<String> pila = new ArrayDeque<>();

        for (String v : g.obtenerVecinos("Planificar")) {
            if (!visitados.contains(v)) {
                dfsTopologico(g, v, visitados, pila);
            }
        }
        pila.push("Planificar");

        List<String> resultado = new ArrayList<>();
        while (!pila.isEmpty()) resultado.add(pila.pop());
        return resultado;
    }

    private static void dfsTopologico(Grafo<String> g, String actual,
                                       Set<String> visitados, Deque<String> pila) {
        visitados.add(actual);
        for (String vecino : g.obtenerVecinos(actual)) {
            if (!visitados.contains(vecino)) {
                dfsTopologico(g, vecino, visitados, pila);
            }
        }
        pila.push(actual);  // Se apila AL TERMINAR de explorar (post-order)
    }

    // ── Reutilizamos detección de ciclo ──
    private static boolean tieneCiclo(Grafo<String> g) {
        Set<String> visitados = new HashSet<>();
        Set<String> enStack = new HashSet<>();
        for (String v : g.obtenerVecinos("Planificar")) {
            if (dfsCiclo(g, v, visitados, enStack)) return true;
        }
        return false;
    }

    private static boolean dfsCiclo(Grafo<String> g, String actual,
                                   Set<String> visitados, Set<String> enStack) {
        visitados.add(actual);
        enStack.add(actual);
        for (String vecino : g.obtenerVecinos(actual)) {
            if (!visitados.contains(vecino)) {
                if (dfsCiclo(g, vecino, visitados, enStack)) return true;
            } else if (enStack.contains(vecino)) return true;
        }
        enStack.remove(actual);
        return false;
    }
}
