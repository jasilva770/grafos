Grafos en Java

Código bien estructurado con clases separadas: Nodo, Arista, Grafo y Main.

Nodo<V> Arista<V> Grafo<V> Main
Desplázate para explorar↓
🏗️

Arquitectura del Proyecto

El proyecto se divide en 4 archivos .java independientes:

📌 Nodo.java

Vértice genérico. Guarda el dato y su lista de aristas.

🔗 Arista.java

Conexión con origen, destino y peso (1.0 por defecto).

🗂️ Grafo.java

Map de nodos, CRUD, BFS, DFS.

▶️ Main.java

Punto de entrada: crea el grafo y lo usa.

Grafo Básico — 4 Archivos Separados

Grafo no dirigido y no ponderado. Usa pestañas para ver cada archivo:

Grafo no dirigido con 6 nodos y 7 aristas
📌 Nodo.java 
import java.util.*;
import java.util.Objects;

/**
 * Clase NODO (vértice) del grafo.
 * Genérica: puede guardar cualquier tipo de dato.
 */
public class Nodo<V> {

    private V dato;
    private List<Arista<V>> aristas;

    public Nodo(V dato) {
        dato = dato;
        aristas = new ArrayList<>();
    }

    // ── GETTERS ──
    public V getDato() { return dato; }

    public List<Arista<V>> getAristas() { return aristas; }

    // ── AGREGAR ARISTA DESDE ESTE NODO ──
    public void agregarArista(Nodo<V> destino, double peso) {
        aristas.add(new Arista<>(dato, destino.getDato(), peso));
    }

    // ── EQUALS Y HASHCODE (basados en el dato) ──
    @Override
    public boolean equals(Object obj) {
        if (dato == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
        return Objects.equals(dato, ((Nodo<?>) obj).dato);
    }

    @Override
    public int hashCode() { return Objects.hashCode(dato); }

    @Override
    public String toString() {
        return "Nodo(" + dato + ")";
    }
}
🔗 Arista.java 
import java.util.Objects;

/**
 * Clase ARISTA (edge) del grafo.
 * Conecta origen → destino con un peso numérico.
 * Para grafos no ponderados, peso = 1.0 por defecto.
 */
public class Arista<V> {

    private V origen;
    private V destino;
    private double peso;

    // Constructor completo
    public Arista(V origen, V destino, double peso) {
        origen = origen;
        destino = destino;
        peso = peso;
    }

    // Constructor sin peso (asume 1.0)
    public Arista(V origen, V destino) {
        origen, destino, 1.0);
    }

    // ── GETTERS ──
    public V getOrigen()  { return origen; }
    public V getDestino() { return destino; }
    public double getPeso() { return peso; }

    public void setPeso(double peso) { peso = peso; }

    @Override
    public boolean equals(Object obj) {
        if (obj == null || getClass() != obj.getClass()) return false;
        Arista<?> o = (Arista<?>) obj;
        return Objects.equals(origen, o.origen)
            && Objects.equals(destino, o.destino);
    }

    @Override
    public String toString() {
        return origen + " → " + destino
            + " (peso: " + peso + ")";
    }
}
🗂️ Grafo.java 
import java.util.*;

/**
 * Implementación de GRAFO con lista de adyacencia.
 * Mapa: dato del vértice → objeto Nodo.
 */
public class Grafo<V> {

    private Map<V, Nodo<V>> nodos;
    private boolean dirigido;

    public Grafo(boolean dirigido) {
        dirigido = dirigido;
        nodos = new HashMap<>();
    }

    // ── CREAR NODO ──
    public void agregarNodo(V dato) {
        nodos.putIfAbsent(dato, new Nodo<>(dato));
    }

    // ── CREAR ARISTA ──
    public void agregarArista(V origen, V destino, double peso) {
        agregarNodo(origen);
        agregarNodo(destino);
        // Arista en un sentido
        nodos.get(origen).agregarArista(nodos.get(destino), peso);
        // Si NO es dirigido: agregar arista inversa
        if (!dirigido) {
            nodos.get(destino).agregarArista(nodos.get(origen), peso);
        }
    }

    // Sobrecarga sin peso (grafo no ponderado)
    public void agregarArista(V origen, V destino) {
        agregarArista(origen, destino, 1.0);
    }

    // ── ELIMINAR NODO ──
    public void eliminarNodo(V dato) {
        if (nodos.containsKey(dato)) {
            // Quitar aristas que apuntan a este nodo
            for (Nodo<V> n : nodos.values()) {
                n.getAristas().removeIf(a -> a.getDestino().equals(dato));
            }
            nodos.remove(dato);
        }
    }

    // ── CONSULTAS ──
    public boolean existeNodo(V dato) {
        return nodos.containsKey(dato);
    }

    public boolean existeArista(V origen, V destino) {
        Nodo<V> n = nodos.get(origen);
        if (n == null) return false;
        return n.getAristas().stream()
            .anyMatch(a -> a.getDestino().equals(destino));
    }

    public int grado(V dato) {
        Nodo<V> n = nodos.get(dato);
        return n != null ? n.getAristas().size() : 0;
    }

    public List<V> obtenerVecinos(V dato) {
        Nodo<V> n = nodos.get(dato);
        if (n == null) return Collections.emptyList();
        return n.getAristas().stream()
            .map(Arista::getDestino).toList();
    }

    public int getNumNodos() { return nodos.size(); }

    // ── BFS (recorrido en anchura) ──
    public List<V> bfs(V inicio) {
        List<V> resultado = new ArrayList<>();
        Set<V> visitados = new HashSet<>();
        Queue<V> cola = new LinkedList<>();
        cola.offer(inicio);
        visitados.add(inicio);

        while (!cola.isEmpty()) {
            V actual = cola.poll();
            resultado.add(actual);
            for (V vecino : obtenerVecinos(actual)) {
                if (!visitados.contains(vecino)) {
                    visitados.add(vecino);
                    cola.offer(vecino);
                }
            }
        }
        return resultado;
    }

    // ── DFS (recorrido en profundidad) ──
    public List<V> dfs(V inicio) {
        List<V> resultado = new ArrayList<>();
        dfsRec(inicio, new HashSet<>(), resultado);
        return resultado;
    }

    private void dfsRec(V actual, Set<V> visitados, List<V> resultado) {
        visitados.add(actual);
        resultado.add(actual);
        for (V vecino : obtenerVecinos(actual)) {
            if (!visitados.contains(vecino))
                dfsRec(vecino, visitados, resultado);
        }
    }

    // ── IMPRIMIR ──
    public void imprimir() {
        System.out.println("=== "
            + (dirigido ? "Grafo Dirigido" : "Grafo No Dirigido")
            + " ===");
        for (Nodo<V> n : nodos.values()) {
            System.out.print("  " + n.getDato() + " → ");
            if (n.getAristas().isEmpty())
                System.out.println("[]");
            else
                System.out.println(n.getAristas());
        }
    }
}
▶️ Main.java 
public class Main {

    public static void main(String[] args) {

        // ══ PASO 1: Crear grafo NO DIRIGIDO ══
        Grafo<String> grafo = new Grafo<>(false);

        // ══ PASO 2: Agregar aristas ══
        // (los nodos se crean automáticamente)
        grafo.agregarArista("A", "B");
        grafo.agregarArista("A", "D");
        grafo.agregarArista("B", "C");
        grafo.agregarArista("B", "E");
        grafo.agregarArista("D", "E");
        grafo.agregarArista("C", "F");
        grafo.agregarArista("E", "F");

        // ══ PASO 3: Imprimir ══
        grafo.imprimir();

        // ══ PASO 4: Consultas ══
        System.out.println("\\nNodos totales: " + grafo.getNumNodos());
        System.out.println("¿Existe nodo B? "
            + grafo.existeNodo("B"));
        System.out.println("¿Existe arista A→B? "
            + grafo.existeArista("A", "B"));
        System.out.println("Grado de B: " + grafo.grado("B"));
        System.out.println("Vecinos de B: "
            + grafo.obtenerVecinos("B"));

        // ══ PASO 5: Recorridos ══
        System.out.println("\\nBFS desde A: "
            + grafo.bfs("A"));
        System.out.println("DFS desde A: "
            + grafo.dfs("A"));
    }
}

📖 ¿Cómo funciona la separación?

  1. Nodo.java — Guarda el dato y su lista de aristas. Es genérico para cualquier tipo.
  2. Arista.java — Guarda origen, destino y peso. Si no se pasa peso, usa 1.0.
  3. Grafo.java — Tiene un Map<V, Nodo<V>> y el flag dirigido. Si es false, duplica cada arista (bidireccional).
  4. Main.java — Solo crea new Grafo<>(false), agrega aristas y llama a los métodos.

Grafo Dirigido

Se reutilizan las mismas 3 clases. Solo cambias new Grafo<>(true).

Flechas = dirección. A→B no implica B→A
▶️ MainDirigido.java 
public class MainDirigido {
    public static void main(String[] args) {

        // ← TRUE = dirigido (aristas en un solo sentido)
        Grafo<String> g = new Grafo<>(true);

        // Red de páginas web
        g.agregarArista("Google", "YouTube");
        g.agregarArista("Google", "Gmail");
        g.agregarArista("YouTube", "Google");
        g.agregarArista("Gmail", "Drive");
        g.agregarArista("YouTube", "Drive");

        g.imprimir();

        System.out.println("\\nGoogle → YouTube: "
            + g.existeArista("Google","YouTube")); // true
        System.out.println("Drive → Gmail: "
            + g.existeArista("Drive","Gmail"));   // false!
    }
}

  Google → [Google → YouTube (peso: 1.0), Google → Gmail (peso: 1.0)]
  YouTube → [YouTube → Google (peso: 1.0), YouTube → Drive (peso: 1.0)]
  Gmail → [Gmail → Drive (peso: 1.0)]
  Drive → []

🔑 La única diferencia

new Grafo<>(true) vs new Grafo<>(false). La clase Grafo internamente decide si duplicar la arista o no. No tocas Nodo ni Arista.

Grafo Ponderado

La clase Arista ya soporta pesos. Solo pasas el tercer parámetro.

Pesos en amarillo: A→B cuesta 5, B→E cuesta 8, etc.
▶️ MainPonderado.java 
import java.util.*;

public class MainPonderado {
    public static void main(String[] args) {

        Grafo<String> g = new Grafo<>(false);

        // Distancias entre ciudades (km)
        g.agregarArista("Madrid", "Barcelona", 621);
        g.agregarArista("Madrid", "Valencia", 355);
        g.agregarArista("Madrid", "Sevilla", 534);
        g.agregarArista("Barcelona", "Valencia", 349);
        g.agregarArista("Sevilla", "Valencia", 650);
        g.agregarArista("Barcelona", "Sevilla", 990);

        g.imprimir();

        // Camino más corto con Dijkstra
        Map<String, Double> d = dijkstra(g, "Madrid");
        d.forEach((c, dist) ->
            System.out.println("  " + c + ": " + dist + " km"));
    }

    // Dijkstra simplificado
    private static Map<String, Double> dijkstra(
            Grafo<String> g, String inicio) {
        Map<String, Double> dists = new HashMap<>();
        Set<String> vis = new HashSet<>();

        for (String v : g.obtenerVecinos(inicio))
            dists.put(v, Double.POSITIVE_INFINITY);
        dists.put(inicio, 0.0);

        PriorityQueue<Map.Entry<String, Double>> cola =
            new PriorityQueue<>(
                Comparator.comparingDouble(Map.Entry::getValue));
        cola.add(new AbstractMap.SimpleEntry<>(inicio, 0.0));

        while (!cola.isEmpty()) {
            String act = cola.poll().getKey();
            if (vis.contains(act)) continue;
            vis.add(act);

            for (String vec : g.obtenerVecinos(act)) {
                double nd = dists.get(act) + 1.0;
                if (nd < dists.getOrDefault(vec, Double.POSITIVE_INFINITY)) {
                    dists.put(vec, nd);
                    cola.add(new AbstractMap.SimpleEntry<>(vec, nd));
                }
            }
        }
        return dists;
    }
}

Grafo Cíclico

Contiene al menos un ciclo: un camino que vuelve al nodo de inicio. Se detecta con DFS.

Ciclo A→B→C→A (líneas gruesas rojas)
▶️ MainCiclico.java 
import java.util.*;

public class MainCiclico {
    public static void main(String[] args) {

        Grafo<String> g = new Grafo<>(true);

        // Crear ciclo: A → B → C → A
        g.agregarArista("A", "B");
        g.agregarArista("B", "C");
        g.agregarArista("C", "A");  // ← Cierra el ciclo!
        g.agregarArista("B", "E");
        g.agregarArista("A", "D");
        g.agregarArista("D", "E");
        g.agregarArista("E", "F");
        g.agregarArista("C", "F");
        g.agregarArista("F", "D");  // ← Segundo ciclo!

        g.imprimir();
        System.out.println("\\n¿Tiene ciclo? " + tieneCiclo(g));
    }

    // Detectar ciclo con DFS
    private static boolean tieneCiclo(Grafo<String> g) {
        Set<String> vis = new HashSet<>();
        Set<String> stk = new HashSet<>();
        for (String v : g.obtenerVecinos("A"))
            if (dfsCiclo(g, v, vis, stk)) return true;
        return false;
    }

    private static boolean dfsCiclo(Grafo<String> g,
            String act, Set<String> vis, Set<String> stk) {
        vis.add(act);
        stk.add(act);
        for (String vec : g.obtenerVecinos(act)) {
            if (!vis.contains(vec)) {
                if (dfsCiclo(g, vec, vis, stk)) return true;
            } else if (stk.contains(vec)) return true;
        }
        stk.remove(act);
        return false;
    }
}
📤 Salida
=== Grafo Dirigido ===
  A → [A → B (peso: 1.0), A → D (peso: 1.0)]
  B → [B → C (peso: 1.0), B → E (peso: 1.0)]
  C → [C → A (peso: 1.0), C → F (peso: 1.0)]
  D → [D → E (peso: 1.0)]
  E → [E → F (peso: 1.0)]
  F → [F → D (peso: 1.0)]

¿Tiene ciclo? true

Grafo Acíclico (DAG)

Sin ciclos. Permite ordenamiento topológico: ordenar nodos respetando dependencias.

DAG: todas las flechas van "hacia abajo". Imposible volver al inicio.
▶️ MainAciclico.java 
import java.util.*;

public class MainAciclico {
    public static void main(String[] args) {

        Grafo<String> g = new Grafo<>(true);

        // Dependencias de un proyecto
        g.agregarArista("Planificar", "Diseñar");
        g.agregarArista("Planificar", "Investigar");
        g.agregarArista("Diseñar", "Programar");
        g.agregarArista("Investigar", "Programar");
        g.agregarArista("Programar", "Probar");
        g.agregarArista("Programar", "Documentar");
        g.agregarArista("Probar", "Lanzar");
        g.agregarArista("Documentar", "Lanzar");

        g.imprimir();

        System.out.println("\\n¿Acíclico? " + !tieneCiclo(g));

        System.out.println("\\nOrden topológico:");
        for (String t : ordenTopologico(g))
            System.out.println("  " + t);
    }

    // Orden topológico con DFS
    private static List<String> ordenTopologico(Grafo<String> g) {
        Set<String> vis = new HashSet<>();
        Deque<String> pila = new ArrayDeque<>();

        for (String v : g.obtenerVecinos("Planificar"))
            if (!vis.contains(v))
                dfsTop(g, v, vis, pila);
        pila.push("Planificar");

        List<String> res = new ArrayList<>();
        while (!pila.isEmpty()) res.add(pila.pop());
        return res;
    }

    private static void dfsTop(Grafo<String> g,
            String act, Set<String> vis, Deque<String> pila) {
        vis.add(act);
        for (String vec : g.obtenerVecinos(act))
            if (!vis.contains(vec))
                dfsTop(g, vec, vis, pila);
        pila.push(act);  // post-order
    }

    // Reutilizamos detección de ciclo
    private static boolean tieneCiclo(Grafo<String> g) {
        Set<String> vis = new HashSet<>();
        Set<String> stk = new HashSet<>();
        for (String v : g.obtenerVecinos("Planificar"))
            if (dfsCiclo(g, v, vis, stk)) return true;
        return false;
    }

    private static boolean dfsCiclo(Grafo<String> g,
            String act, Set<String> vis, Set<String> stk) {
        vis.add(act); stk.add(act);
        for (String vec : g.obtenerVecinos(act)) {
            if (!vis.contains(vec)) {
                if (dfsCiclo(g, vec, vis, stk)) return true;
            } else if (stk.contains(vec)) return true;
        }
        stk.remove(act);
        return false;
    }
}
